345

348

358  while(notelist_curr != NULL) {

361  /* update */
362  note_curr->update(ticks_delta, pixels_per_tick);

367  timing_window = MISS;
368  } else if(note_curr_ticks_to_hit <= (int16_t)(-2)) {
369  timing_window = GOOD_LATE;

371  timing_window = GREAT;
372  } else if(note_curr_ticks_to_hit <= (int16_t)(6)) {

386  }

391  break;

395  uint8_t bit = 0;

397  scoreboard.greats++;
398  scoreboard.combo++;
399  scoreboard.total_score += SCORE_GAIN_GREAT;
400  scored_poptime = ticks_per_beat*2U;
401  scored_last = timing_window;
402  note_curr->grade = timing_window;
403  while (button >>= 1) {
404  bit++;
405  }
406  hitmarks[bit] = 10;
407  }
408  break;
409  case GOOD_LATE:
410  case GOOD_EARLY:
411  if(arduboy.justPressed(note_curr->button)) {
412  uint8_t button = note_curr->button;
413  uint8_t bit = 0;
414  life_bar += (int8_t)LIFE_GAIN_GOOD;
415  scoreboard.goods++;
416  scoreboard.combo++;
417  scoreboard.total_score += SCORE_GAIN_GOOD;
418  scored_poptime = ticks_per_beat*2U;
419  scored_last = timing_window;
420  note_curr->grade = timing_window;
421  while (button >>= 1) {
422  bit++;
423  }
424  hitmarks[bit] = 10;
425  }
426  break;
427  default:
428  break;
429  }
430

432  {
433  navi_poptime = ticks_per_beat*4U;

435  }

644 uint8_t rhythm_manager::parse_chart_smlike(const uint8_t **chart_index, uint16_t *ticks_parsed)
645 {
646  /* local variables */
647  uint8_t action_curr;
648  uint8_t lines_in_measure;
649  uint8_t notes_in_measure;
650  const uint8_t *readahead_index;
651  notelist * notelist_curr;
652 
653  if((chart_index == NULL) || (ticks_parsed == NULL)) {
654  /* we have no valid chart, so return */
655  return CHART_PARSED;
656  }
657 
658  #ifdef DEBUG_SERIAL
659  Serial.println("chart good");
660  #endif
661 
662  if (notelist_free_head == NULL) {
663  /* we have no space for new notes, so return */
664  return CHART_STALLED;
665  }
666 
667  #ifdef DEBUG_SERIAL
668  Serial.println("at least one note free");
669  #endif
670 
671  while (pgm_read_byte(*chart_index) != PATTERN_EOF) {
672  /* determine how many lines in the next measure */
673  readahead_index = *chart_index;
674  lines_in_measure = 0;
675  notes_in_measure = 0;
676  /* start reading line by line */
677  action_curr = pgm_read_byte((readahead_index)++);
678  while (action_curr != PATTERN_EOM) {
679  #ifdef DEBUG_SERIAL

681  Serial.println(action_curr);
682  Serial.print("readahead metadata: ");
683  Serial.println(metadata_curr);
684  #endif
685  lines_in_measure++;
686  while (action_curr) {
687  notes_in_measure += action_curr & 1U;
688  action_curr >>= 1;
689  }
690  action_curr = pgm_read_byte((readahead_index)++);
691  }
692 
693  #ifdef DEBUG_SERIAL
694  Serial.print("lines in measure: ");
695  Serial.println(lines_in_measure);
696  Serial.print("notes in measure: ");
697  Serial.println(notes_in_measure);
698  #endif
699 
700  /* see if we have enough open notes for the whole measure */
701  notelist_curr = notelist_free_head;
702  for (uint8_t cnt = 1U; cnt < notes_in_measure; cnt++) {
703  notelist_curr = notelist_curr->get_prev();
704  if(NULL == notelist_curr) {
705  #ifdef DEBUG_SERIAL
706  Serial.print("ran out of notes?");
707  #endif
708  return CHART_STALLED;
709  }
710  }
711 
712  #ifdef DEBUG_SERIAL
713  Serial.println("enough open notes");
714  #endif
715 
716  /* if we have enough open notes, read in the whole measure */
717  /* start reading line by line */
718  action_curr = pgm_read_byte((*chart_index)++);
719  while(action_curr != PATTERN_EOM) {
720  #ifdef DEBUG_SERIAL
721  Serial.println(action_curr);
722  Serial.println(metadata_curr);
723  #endif
724 
725  parse_action(action_curr, ticks_parsed);
726 
727  /* we finished loading the action, increment timeline by the right amount */
728  *ticks_parsed += ((ticks_per_beat*4U)/lines_in_measure);
729 
730  /* read the next line */
731  action_curr = pgm_read_byte((*chart_index)++);
732  }
733 
734  /* we finished loading the measure, loop for more measures */
735  }
736 
737  /* we finished loading the whole chart, return */
738  return CHART_PARSED;

742 {

786 uint8_t rhythm_manager::parse_action(uint8_t action_curr, uint16_t *ticks_parsed)
787 {
788  uint8_t offset_curr;
789  uint8_t button_curr;
790  notelist * notelist_curr;
791  note * note_curr;
792 
793  /* determine which notes are in the action */
794  while (action_curr) {
795  if (action_curr & A_BUTTON) {
796  offset_curr = SLOT5_Y;
797  button_curr = A_BUTTON;
798  action_curr = action_curr & (~A_BUTTON);
799  } else if (action_curr & B_BUTTON) {
800  offset_curr = SLOT6_Y;
801  button_curr = B_BUTTON;
802  action_curr = action_curr & (~B_BUTTON);
803  } else if (action_curr & UP_BUTTON) {
804  offset_curr = SLOT3_Y;
805  button_curr = UP_BUTTON;
806  action_curr = action_curr & (~UP_BUTTON);
807  } else if (action_curr & DOWN_BUTTON) {
808  offset_curr = SLOT2_Y;
809  button_curr = DOWN_BUTTON;
810  action_curr = action_curr & (~DOWN_BUTTON);
811  } else if (action_curr & LEFT_BUTTON) {
812  offset_curr = SLOT1_Y;
813  button_curr = LEFT_BUTTON;
814  action_curr = action_curr & (~LEFT_BUTTON);
815  } else if (action_curr & RIGHT_BUTTON) {
816  offset_curr = SLOT4_Y;
817  button_curr = RIGHT_BUTTON;
818  action_curr = action_curr & (~RIGHT_BUTTON);
819  } else {
820  /* ERROR, should not ever get here */
821  return CHART_PARSED;
822  }
823 
824  /* if we reach here, we're initing a note */
825  notelist_curr = notelist_free_head;
826  note_curr = &(notelist_curr->data);
827  note_curr->init(offset_curr, ((*ticks_parsed)-ticks_curr)*pixels_per_tick, *ticks_parsed, button_curr);
828  /* remove from free list */
829  notelist_free_head = notelist_curr->get_prev();
830  notelist_curr->remove();
831 
832  /* double link to used list */
833  notelist_curr->set_prev(notelist_used_head);
834  if(notelist_used_head != NULL) {
835  notelist_used_head->set_next(notelist_curr);
836  }
837  notelist_used_head = notelist_curr;
838  }
839 
840  return 0U;
841 }

843 uint8_t rhythm_manager::generate_step(void)
844 {
845  uint8_t action = (uint8_t)_BV((uint8_t)(random(6U)+2U));
846  if(!random(10U)) {
847  /* 10% chance of randomly making a jump */
848  action |= (uint8_t)_BV((uint8_t)(random(6U)+2U));
849  }
850  /* ed's big thumb bad jump filter */
851  switch (action)
852  {
853  case (UP_BUTTON | LEFT_BUTTON): /* fallthrough */
854  case (UP_BUTTON | RIGHT_BUTTON): /* fallthrough */
855  case (DOWN_BUTTON | LEFT_BUTTON): /* fallthrough */
856  case (DOWN_BUTTON | RIGHT_BUTTON): /* fallthrough */
857  case (A_BUTTON | B_BUTTON):
858  /* regenerate a good single step instead */
859  action = (uint8_t)_BV((uint8_t)(random(6U)+2U));
860  break;
861  default:
862  /* we good, jump on */
863  break;
864  }
865  return action;
