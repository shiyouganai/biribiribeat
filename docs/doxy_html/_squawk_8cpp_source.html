139 // Look up or generate waveform for ProTracker vibrato/tremolo oscillator
140 static int8_t do_osc(pto_t *p_osc) {
141  int8_t sample = 0;
142  int16_t mul;
143  switch(p_osc->mode & 0x03) {
144  case 0: // Sine
145  sample = pgm_read_byte(&sine_tbl[(p_osc->offset) & 0x1F]);
146  if(p_osc->offset & 0x20) sample = -sample;
147  break;
148  case 1: // Saw
149  sample = -(p_osc->offset << 2);
150  break;
151  case 2: // Square
152  sample = (p_osc->offset & 0x20) ? 127 : -128;
153  break;
154  case 3: // Noise (random)
155  sample = rand();
156  break;
157  }
158  mul = sample * LO4(p_osc->fxp);
159  p_osc->offset = (p_osc->offset + HI4(p_osc->fxp));
160  return mul >> 6;
161 }

173 // Calculates and returns glissando period
174 // Essentially snaps a sliding frequency to the closest note
175 static inline uint16_t glissando(uint8_t ch) {
176  uint8_t n;
177  uint16_t period_h, period_l;
178  for(n = 0; n != 47; n++) {
179  period_l = pgm_read_word(&period_tbl[n]);
180  period_h = pgm_read_word(&period_tbl[n + 1]);
181  if(fxm[ch].period < period_l && fxm[ch].period >= period_h) {
182  if(period_l - fxm[ch].period <= fxm[ch].period - period_h) {
183  period_h = period_l;
184  }
185  break;
186  }
187  }
188  return period_h;

311 // Decrunches a 9 byte row into a useful data
312 static void decrunch_row() {
313  uint8_t data;
314 
315  // Initial decrunch
316  stream->seek(stream_base + ((order[ix_order] << 6) + ix_row) * 9);
317  data = stream->read(); cel[0].fxc = data << 0x04;
318  cel[1].fxc = data & 0xF0;
319  data = stream->read(); cel[0].fxp = data;
320  data = stream->read(); cel[1].fxp = data;
321  data = stream->read(); cel[2].fxc = data << 0x04;
322  cel[3].fxc = data >> 0x04;
323  data = stream->read(); cel[2].fxp = data;
324  data = stream->read(); cel[3].fxp = data;
325  data = stream->read(); cel[0].ixp = data;
326  data = stream->read(); cel[1].ixp = data;
327  data = stream->read(); cel[2].ixp = data;
328 
329  // Decrunch extended effects
330  if(cel[0].fxc == 0xE0) { cel[0].fxc |= cel[0].fxp >> 4; cel[0].fxp &= 0x0F; }
331  if(cel[1].fxc == 0xE0) { cel[1].fxc |= cel[1].fxp >> 4; cel[1].fxp &= 0x0F; }
332  if(cel[2].fxc == 0xE0) { cel[2].fxc |= cel[2].fxp >> 4; cel[2].fxp &= 0x0F; }
333 
334  // Decrunch cell 3 ghetto-style
335  cel[3].ixp = ((cel[3].fxp & 0x80) ? 0x00 : 0x7F) | ((cel[3].fxp & 0x40) ? 0x80 : 0x00);
336  cel[3].fxp &= 0x3F;
337  switch(cel[3].fxc) {
338  case 0x02:
339  case 0x03: if(cel[3].fxc & 0x01) cel[3].fxp |= 0x40; cel[3].fxp = (cel[3].fxp >> 4) | (cel[3].fxp << 4); cel[3].fxc = 0x70; break;
340  case 0x01: if(cel[3].fxp & 0x08) cel[3].fxp = (cel[3].fxp & 0x07) << 4; cel[3].fxc = 0xA0; break;
341  case 0x04: cel[3].fxc = 0xC0; break;
342  case 0x05: cel[3].fxc = 0xB0; break;
343  case 0x06: cel[3].fxc = 0xD0; break;
344  case 0x07: cel[3].fxc = 0xF0; break;
345  case 0x08: cel[3].fxc = 0xE7; break;
346  case 0x09: cel[3].fxc = 0xE9; break;
347  case 0x0A: cel[3].fxc = (cel[3].fxp & 0x08) ? 0xEA : 0xEB; cel[3].fxp &= 0x07; break;
348  case 0x0B: cel[3].fxc = (cel[3].fxp & 0x10) ? 0xED : 0xEC; cel[3].fxp &= 0x0F; break;
349  case 0x0C: cel[3].fxc = 0xEE; break;
350  }
351 
352  // Apply generic effect parameter memory
353  uint8_t ch;
354  cel_t *p_cel = cel;
355  fxm_t *p_fxm = fxm;
356  for(ch = 0; ch != 4; ch++) {
357  uint8_t fx = p_cel->fxc;
358  if(fx == 0x10 || fx == 0x20 || fx == 0xE1 || fx == 0xE2 || fx == 0x50 || fx == 0x60 || fx == 0xA0) {
359  if(p_cel->fxp) {
360  p_fxm->param = p_cel->fxp;
361  } else {
362  p_cel->fxp = p_fxm->param;
363  }
364  }
365  p_cel++; p_fxm++;
366  }
367 }

503 // Progress module by one tick
504 void __attribute__((used)) squawk_playroutine() {
505  static bool lockout = false;
506 
507  if(!order_count) return;
508 
509  // Protect from re-entry via ISR
510  cli();
511  if(lockout) {
512  sei();
513  return;
514  }
515  lockout = true;
516  sei();
517 
518  // Increment total tick counter
519  tick_counter++;
520  // If we inserted blank ticks before song, decrement and return
521  if((0 == ix_row) &&
522  (0 == ix_order) &&
523  count_in) {
524  count_in--;
525  lockout = false;
526  return;
527  }
528 
529  // Handle row delay
530  if(row_delay) {
531  if(tick == 0) row_delay--;
532  // Advance tick
533  if(++tick == speed) tick = 0;
534  } else {
535 
536  // Quick pointer access
537  fxm_t *p_fxm = fxm;
538  osc_t *p_osc = osc;
539  cel_t *p_cel = cel;
540 
541  // Temps
542  uint8_t ch, fx, fxp;
543  bool pattern_jump = false;
544  uint8_t ix_period;
545 
546  for(ch = 0; ch != 4; ch++) {
547  uint8_t temp;
548 
549  // Local register copy
550  fx = p_cel->fxc;
551  fxp = p_cel->fxp;
552  ix_period = p_cel->ixp;
553 
554  // If first tick
555  if(tick == (fx == 0xED ? fxp : 0)) {
556 
557  // Reset volume
558  if(ix_period & 0x80) p_osc->vol = p_fxm->volume = 0x24;
559  #if 1
560  if((ix_period & 0x80) && ch == 3) {
561  p_osc->vol = p_fxm->volume = 0x14;
562  }
563  #endif
564 
565  if((ix_period & 0x7F) != 0x7F) {
566 
567  // Reset oscillators (unless continous flag set)
568  if((p_fxm->vibr.mode & 0x4) == 0x0) p_fxm->vibr.offset = 0;
569  if((p_fxm->trem.mode & 0x4) == 0x0) p_fxm->trem.offset = 0;
570 
571  // Cell has note
572  if(fx == 0x30 || fx == 0x50) {
573 
574  // Tone-portamento effect setup
575  p_fxm->port_target = pgm_read_word(&period_tbl[ix_period & 0x7F]);
576  } else {
577 
578  // Set required effect memory parameters
579  p_fxm->period = pgm_read_word(&period_tbl[ix_period & 0x7F]);
580 
581  // Start note
582  if(ch != 3) p_osc->freq = FREQ(p_fxm->period);
583 
584  }
585  }
586 
587  // Effects processed when tick = 0
588  switch(fx) {
589  #ifdef EFFECT_30
590  case 0x30: // Portamento
591  if(fxp) p_fxm->port_speed = fxp;
592  break;
593  #endif
594  #ifdef EFFECT_B0
595  case 0xB0: // Jump to pattern
596  ix_nextorder = (fxp >= order_count ? 0x00 : fxp);
597  ix_nextrow = 0;
598  pattern_jump = true;
